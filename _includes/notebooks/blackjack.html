import random

#define a class to simulate a hand and draws
class hand:
    def __init__(self):
        self.cards = []
        self.fv = 0
        
    def state(self):
        if len(self.cards)>0:
            self.fv = sum([min(10, x) for x in self.cards])
        useable_ace = False
        if 1 in self.cards:
            if self.fv+10<=21:
                useable_ace = True
                self.fv+=10
        return [self.fv, useable_ace]
    
    def draw(self):
        n = random.randint(1, 13)
        self.cards.append(n)
        self.state()
    
    def firstValue(self):
        if len(self.cards)>0:
            return min(self.cards[0], 10)
        return None

import numpy as np

q_table = {} #value function
n_table = {} #count of how many times each tuple has occured
w_table = {} #cumulative counts for win/draw/loss for each tuple

#i indicates my own card values, b indicates whether I have a usable ace, and d indicates the value of the dealer's hand.
for i in range(12, 21):
    for b in [True, False]:
        for d in range(1, 11):
            q_table[(i, b, d)] = [0.0, 0.0]
            n_table[(i, b, d)] = np.array([0,0])
            w_table[(i, b, d)] = np.zeros([2,3])

#we simulate epoch number of random games
eps = 0.2
epoch = 5000000
#random.seed()
eps_greedy = True

for x in range(epoch):
    #0 means hit, 1 means stick
    action = 0
    state_seqs = []
    player = hand()
    while player.fv<12:
        player.draw()

    dealer = hand()
    dealer.draw()
    dealer.draw()
    
    #print(player.fv, dealer.fv)
                      
    if player.fv==21:
        if dealer.fv!=21:
            reward = 1
        else:
            reward = 0

    else:
        while action!=1:
            cur_state = tuple(player.state()+[dealer.firstValue()])
            
            m = q_table[cur_state]
            action = np.argmax(m)
            
            if eps_greedy:
                if random.random()<eps:
                    action = random.randint(0,1)
            
            if player.fv<21:
                cur_sa = (cur_state, action)
                state_seqs.append(cur_sa)
            
            if action==0:
                player.draw()
                if player.fv>21:
                    reward=-1
                    break
                if player.fv==21:
                    break

        if player.fv<=21:
            while dealer.fv<17:
                dealer.draw()
            if (dealer.fv>21 or player.fv>dealer.fv):
                reward=1
            elif dealer.fv==player.fv:
                reward=0
            else:
                reward=-1
                
    for sa in state_seqs:
        st, ac = sa
        n_table[st][ac]+=1
        q_table[st][ac]+=(1.0/n_table[st][ac])*(reward-q_table[st][ac])
        w_table[st][ac][reward]+=1
        #print(sa, q_table[st][ac])
